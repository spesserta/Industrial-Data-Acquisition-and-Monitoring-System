工业上位机监控程序（SCADA），就像工厂里的 “中控室大脑，他能一边通过串口和 PLC（现场控制器）通信，采集设备的实时数据；一边把数据显示在电脑界面上，让工人/工程师能看到设备状态、控制设备；同时把重要数据存到数据库，方便以后查记录、做报表。
各部分的职责如下：

- 界面层（主窗口/控件）：用户能看到的窗口、按钮、图表，只负责显示数据和接收操作，不处理逻辑。
- ViewModel：界面和后台的 “翻译官”，负责数据绑定，把后台数据传给界面、把界面操作传给后台。
- Model：数据的 “模板 / 容器”，定义设备数据的格式（比如温度、状态），供所有层统一使用。
- BLL 业务逻辑层：程序的 “规则中心”，处理报警判断、数据校验等业务规则，承上启下。
- DAL 数据访问层：只负责读写数据库、配置文件，和数据存储打交道，不处理业务逻辑。
- 全局监控 + Modbus 采集：和 PLC 通信的 “手脚”，负责初始化程序、定时采集 PLC 数据，是程序和硬件的桥梁。

一般来说项目是从下往上写的。

## DAL层设计

在DAL层设计一个数据库访问的类DataAssess.cs，然后在这个类里面封装好数据库相关的操作

```c#
using System;
using System.Collections.Generic;
using System.Configuration;
using System.Data;
using System.Data.SqlClient;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace WPFApp.DAL
{
    public class DataAssess
    {
        string dbConfig = ConfigurationManager.ConnectionStrings["db_config"].ToString(); //获取数据库的连接地址
        //定义几个基本的SQL连接变量
        SqlConnection conn;
        SqlCommand comm;
        SqlDataAdapter adapter;
        SqlTransaction trans;


        private void Dispose()  //做销毁
        {
            if(adapter != null)
            {
                adapter.Dispose();
                adapter = null;
            }
            if(comm != null)
            {
                comm.Dispose();
                comm = null;
            }
            if(trans!=null)
            {
                trans.Dispose();
                trans = null;
            }
            if(conn != null)
            {
                conn.Close();
                conn.Dispose();
                conn = null;
            }
        }

        private DataTable GetDatas(string sql)  //获取数据库的数据
        {
            DataTable dt = new DataTable();
            try
            {
                conn = new SqlConnection(dbConfig);
                conn.Open();
                adapter = new SqlDataAdapter(sql, conn);
                adapter.Fill(dt);
            }
            catch(Exception ex)
            {
                throw ex;
            }
            finally
            {
                this.Dispose();
            }

            return dt;
        }
        public DataTable GetStorageArea()  //获取存储区的相关数据
        {
            string strSql = "select * from storage_area";
            return this.GetDatas(strSql);
        }
        public DataTable GetDevices()   //获取所有的设备信息
        {
            string strSql = "select * from devices";
            return this.GetDatas(strSql);
        }
        public DataTable GetMonitorValues() //获取检测器的信息
        {
            string strSql = "select * from Monitor_values order by d_id , value_id";
            return this.GetDatas(strSql);
        }


    }
    
}
```

```c#
using System;
using System.Collections.Generic;
using System.Configuration;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Data;
using System.Windows;
using Communication;

namespace WPFApp.BLL
{
    public class IndustrialBLL
    {
        //获取串口信息
        public DataResult<SerialInfo> InitSerialInfo()
        {
            //返回结果对象，用来存放【获取状态】+【异常信息】+【需要的数据】
            DataResult<SerialInfo> result = new DataResult<SerialInfo>();
            //默认先标记为：执行失败
            result.State = false;

            try
            {
                SerialInfo si = new SerialInfo();
                //从App.config里读取串口名、波特率、数据位、校验位、停止位
                si.PortName = ConfigurationManager.AppSettings["port"].ToString();
                si.BaudRate = int.Parse(ConfigurationManager.AppSettings["baud"].ToString());
                si.DataBit = int.Parse(ConfigurationManager.AppSettings["data_bit"].ToString());
                si.parity = (System.IO.Ports.Parity)Enum.Parse(typeof(System.IO.Ports.Parity), ConfigurationManager.AppSettings["parity"].ToString(), true);
                si.stopBits = (System.IO.Ports.StopBits)Enum.Parse(typeof(System.IO.Ports.StopBits), ConfigurationManager.AppSettings["stop_bit"].ToString(), true);
                //所有参数读取成功
                result.State = true;
                //将串口信息打包好放入si里面
                result.Data = si;

            }
            catch (Exception ex)
            {
                result.Message = ex.Message;
            }
            // 把最终结果result返回给调用方
            return result;
        }
    }
}
```

```c#
{
    DataResult<List<StorageModel>> result = new DataResult<List<StorageModel>>();
    try
    {
        //从数据库获取所有存储区信息
        var sa = da.GetStorageArea();
        //所有参数读取成功
        result.State = true;
        //将存储区信息打包好放入里面
        result.Data = (from q in sa.AsEnumerable()   //LINQ语句，把数据库查出来的表数据转成程序能用的对象列表
                       select new StorageModel
                       {
                           id = q.Field<string>("id"),   //存储区编号
                           SlaveAddress = q.Field<Int32>("slave_add"),  //从站地址
                           FuncCode = q.Field<string>("func_code"),  //Modbus 功能码
                           StartAddress = int.Parse(q.Field<string>("start_reg")),  //起始寄存器地址
                           Length = int.Parse(q.Field<string>("length"))  //读多少个地址
                       }).ToList();
    }
    catch (Exception ex)
    {
        result.Message=ex.Message;
    }
    return result;
}
```

```c#
            set { _currentValue = value;
                if (IsAlarm)
                {
                    string msg = ValueDesc;
                    MonitorValueState state = MonitorValueState.OK;

                    if (value < LoLoAlarm)
                    {
                        state = MonitorValueState.LoLo;
                        msg += "极低";
                    }
                    else if (value < LowAlarm)
                    {
                        state = MonitorValueState.Low;
                        msg += "过低";
                    }
                    else if (value > HiHiAlarm)
                    {
                        state = MonitorValueState.HiHi;
                        msg += "极高";
                    }
                    else if (value > HighAlarm)
                    {
                        state = MonitorValueState.High;
                        msg += "过高";
                    }
                    //每当设置一次值的时候就会触发一次这个委托，过一下每一次值的变化
                    ValueStateChange(state,msg+"。当前值："+value.ToString(),ValueId);
                }
            }
        }
        public string ValueDesc { get; set; }
        
    }
}
```

```c#
public DataResult<List<DevicesModel>> InitDevices()  //初始化设备信息
{
    //返回结果对象，用来存放【状态】+【信息】+【数据】
    DataResult<List<DevicesModel>> result = new DataResult<List<DevicesModel>>();
    try
    {
        var devices = da.GetDevices();   //获取数据库的所有设备信息
        var monitorValues = da.GetMonitorValues(); //获取数据库的监测器信息

        List<DevicesModel> deviceList = new List<DevicesModel>();  //拿来装所有的设备信息

        foreach (var dr in devices.AsEnumerable())
        {
            //给当前设备创建一个模型对象
            DevicesModel dModel = new DevicesModel();
            //把这个设备加入最终列表
            deviceList.Add(dModel);
            //给设备赋值：设备ID、设备名称
            dModel.DeviceID = dr.Field<string>("d_id");
            dModel.DeviceName = dr.Field<string>("d_name");
            //遍历【属于当前设备】的所有监控点
            foreach (var mv in monitorValues.AsEnumerable().Where(m => m.Field<string>("id") ==
            dModel.DeviceID))
            {
                //创建一个监控点模型
                MonitorValueModel mvm = new MonitorValueModel();
                //把监控点加入当前设备的监控列表
                dModel.MonitorValueList.Add(mvm);
                //给监控点赋值（从数据库读出来）
                mvm.ValueId = mv.Field<string>("value_id");
                mvm.ValueName = mv.Field<string>("value_name");
                mvm.StorageAreaID = mv.Field<string>("s_area_id");
                mvm.StartAddress = mv.Field<Int32>("address");
                mvm.DataType = mv.Field<string>("data_type");
                mvm.IsAlarm = mv.Field<Int32>("is_alarm") == 1;
                mvm.ValueDesc = mv.Field<string>("description");
                mvm.Unit = mv.Field<string>("unit");

                // 读取4个报警阈值（为空就给0.0）
                var column = mv.Field<string>("alarm_lolo");
                mvm.LoLoAlarm = column == null ? 0.0 : double.Parse(column);  //空的话返回0.0否则正常返回
                column = mv.Field<string>("alarm_low");
                mvm.LowAlarm = column == null ? 0.0 : double.Parse(column);
                column = mv.Field<string>("alarm_high");
                mvm.HighAlarm = column == null ? 0.0 : double.Parse(column);
                column = mv.Field<string>("alarm_hihi");
                mvm.HiHiAlarm = column == null ? 0.0 : double.Parse(column);
                //绑定“报警状态变化”的动作，当监控点数值越限时，会执行这里的代码
                mvm.ValueStateChange = new Action<Base.MonitorValueState, string, string>((state, msg, value_id) =>
                {
                    //先找：当前监控点的报警是否已经在列表里了
                    var index = dModel.WarningMessageList.ToList().FindIndex(w => w.ValueId == value_id);
                    //如果找到了，先删掉旧的（避免重复报警）
                    if (index > -1)
                      dModel.WarningMessageList.RemoveAt(index);

                     if (state != Base.MonitorValueState.OK)
                     {
                        dModel.isWarning = true;  // 设备标记为报警状态
                        // 把报警信息加入列表
                        dModel.WarningMessageList.Add(new WarningMessageModel { ValueId = value_id,
                               Message = msg});
                      }
                    //最后判断：有没有报警信息
                    var ss = dModel.WarningMessageList.Count > 0;
                    //同步设备的报警状态（有消息=报警，无消息=正常）
                    if (dModel.isWarning != ss)
                         dModel.isWarning = ss;
                });
            }
        }
        result.State = true;
        result.Data=deviceList; 
    }
    catch
    {

    }
    return result;
}
```
